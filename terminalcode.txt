ls -/- lista tudo
cd // cd /home/estudante... -/-  escolhe aonde o terminal está
cd .. -/-  volta um diretorio
mkdir -/-  cria uma pasta
pwd -/-  diz aonde voce está
clean -/-  limpa o terminal
code -/- abre a pasta no vsc
history -/- mostra o historico dos codigos
git config --global user.name "nome" -/- define o nome do usuario que fez as alterações
git config --global user.email "email" -/- define o email do usuario 
git config -l -/- lista as coisas
git status -/- aponta os status do git
git init -/- inicia o git
git add "oq vc quer" -/- adiciona o que voce quer no commit
git add . -/- adiciona tudo
git commit -m "comentario" -/- commita tudo que estava no add
"comentario" = esse commite faz o que?
git commit -m "comentario" -/-amend -///- muda o comentario do seu ultimo commit
git restore -/-staged arquivo que voce nao quer comitar
git reset HEAD ~1 -/- restaura o ultimo commit, como se n fosse feito
